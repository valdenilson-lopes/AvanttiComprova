📚 Documentação do Banco de Dados - Avantti Comprova (Supabase/PostgreSQL)
🎯 Visão Geral
O banco de dados do Avantti Comprova gerencia:

Empresas: Cadastro, licença (comprova), limite de usuários

Motoristas: Vinculados às empresas, com status ativo/inativo

Sessões: Controle de login, token, deviceId, timeout e substituição por outro dispositivo

🗂️ Estrutura das Tabelas (Árvore do Banco)

📦 Banco de Dados (Supabase PostgreSQL)
│
├── 📁 Tabelas
│   ├── empresas              # Cadastro de empresas e licenças
│   ├── motoristas            # Motoristas vinculados às empresas
│   └── sessoes               # Sessões ativas/inativas dos motoristas
│
├── 📁 Índices
│   ├── idx_empresas_licenca
│   ├── idx_motoristas_empresa
│   ├── idx_sessoes_ativa
│   └── unica_sessao_ativa    # 🔥 Garante apenas uma sessão ativa por motorista
│
├── 📁 Funções (RPC)
│   ├── login()               # Login unificado com deviceId
│   ├── validar_sessao()      # Valida token + timeout + substituição
│   ├── logout()              # Logout manual por CNPJ/motorista
│   ├── logout_by_token()     # Logout por token (alternativo)
│   ├── atualizar_ultimo_acesso()  # Atualiza timestamp de atividade
│   ├── get_current_cnpj()    # 🔥 Usado nas políticas RLS
│   └── limpar_sessoes_inativas()  # Manutenção (opcional)
│
├── 📁 RLS (Row Level Security)
│   ├── empresas → política "Usuarios veem apenas sua empresa"
│   ├── motoristas → política "Usuarios veem motoristas da sua empresa"
│   └── sessoes → políticas de SELECT e UPDATE
│
└── 📁 Permissões
    └── anon → apenas EXECUTE nas funções (acesso direto às tabelas REVOKED)
📄 1. TABELAS (Versão Final)
1.1 Tabela empresas
sql
CREATE TABLE IF NOT EXISTS public.empresas
(
    cnpj character varying(14) NOT NULL,
    nome_fantasia character varying(255) NOT NULL,
    status boolean DEFAULT true,
    codfilial character varying(2) NOT NULL DEFAULT '01',
    licenca_comprova character varying(2) NOT NULL DEFAULT '01',
    limite_usuarios_comprova integer NOT NULL,
    comprova_expira_em timestamp without time zone NOT NULL,
    comprova_ativa boolean DEFAULT true,
    CONSTRAINT empresas_pkey PRIMARY KEY (cnpj)
);

-- Índice para consultas de licença
CREATE INDEX IF NOT EXISTS idx_empresas_licenca
    ON public.empresas USING btree
    (cnpj ASC, comprova_ativa ASC, comprova_expira_em ASC);
1.2 Tabela motoristas
sql
CREATE TABLE IF NOT EXISTS public.motoristas
(
    id integer GENERATED BY DEFAULT AS IDENTITY,
    codfilial character varying(2),
    codigo_motorista character varying(50),
    nome character varying(255),
    ativo boolean DEFAULT true,
    cnpj character varying(14),
    CONSTRAINT motoristas_pkey PRIMARY KEY (id),
    CONSTRAINT motoristas_cnpj_fkey FOREIGN KEY (cnpj)
        REFERENCES public.empresas (cnpj)
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);

CREATE INDEX IF NOT EXISTS idx_motoristas_empresa
    ON public.motoristas USING btree (cnpj ASC, nome ASC);
1.3 Tabela sessoes
sql
CREATE TABLE IF NOT EXISTS public.sessoes
(
    id integer GENERATED BY DEFAULT AS IDENTITY,
    codfilial character varying(2),
    motorista character varying(50),
    device_id character varying(255),
    token character varying(255),
    ativo boolean DEFAULT true,
    data_login timestamp without time zone,
    ultimo_acesso timestamp without time zone,
    data_encerramento timestamp without time zone,
    motivo_encerramento character varying(255),
    cnpj character varying(14),
    ip character varying(50),
    CONSTRAINT sessoes_pkey PRIMARY KEY (id)
);

-- Índice para busca de sessões ativas
CREATE INDEX IF NOT EXISTS idx_sessoes_ativa
    ON public.sessoes USING btree (cnpj ASC, motorista ASC)
    WHERE (ativo = true);

-- 🔥 Índice Único: Garante que um motorista não tenha duas sessões ativas simultâneas
CREATE UNIQUE INDEX IF NOT EXISTS unica_sessao_ativa
    ON public.sessoes USING btree (cnpj ASC, motorista ASC)
    WHERE (ativo = true);

⚙️ 3. FUNÇÕES (Versão Final)
3.1 Função login()
CREATE OR REPLACE FUNCTION login(
    p_cnpj TEXT,
    p_motorista TEXT,
    p_device_id TEXT
)
RETURNS JSONB
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_empresa RECORD;
    v_motorista RECORD;
    v_sessao_existente RECORD;
    v_sessoes_ativas INTEGER;
    v_token TEXT;
    v_resultado JSONB;
BEGIN
    -- ==============================================
    -- 1. VALIDAR EMPRESA
    -- ==============================================
    SELECT * INTO v_empresa
    FROM empresas
    WHERE cnpj = p_cnpj;
    
    IF v_empresa IS NULL THEN
        RAISE LOG 'Login falhou - Empresa não encontrada: %', p_cnpj;
        RETURN json_build_object('error', 'Empresa não encontrada');
    END IF;
    
    -- ==============================================
    -- 2. VALIDAR STATUS DA EMPRESA
    -- ==============================================
    IF NOT v_empresa.status THEN
        RAISE LOG 'Login falhou - Empresa bloqueada: %', p_cnpj;
        RETURN json_build_object('error', 'Empresa bloqueada');
    END IF;
    
    -- ==============================================
    -- 3. VALIDAR LICENÇA
    -- ==============================================
    IF NOT v_empresa.comprova_ativa THEN
        RAISE LOG 'Login falhou - Licença desativada: %', p_cnpj;
        RETURN json_build_object('error', 'Licença desativada');
    END IF;
    
    IF v_empresa.comprova_expira_em < NOW() THEN
        RAISE LOG 'Login falhou - Licença expirada: % em %', p_cnpj, v_empresa.comprova_expira_em;
        RETURN json_build_object('error', 'Licença expirada em ' || to_char(v_empresa.comprova_expira_em, 'DD/MM/YYYY'));
    END IF;
    
    -- ==============================================
    -- 4. VALIDAR MOTORISTA
    -- ==============================================
    SELECT * INTO v_motorista
    FROM motoristas
    WHERE codigo_motorista = p_motorista
      AND cnpj = p_cnpj
      AND ativo = true;
    
    IF v_motorista IS NULL THEN
        RAISE LOG 'Login falhou - Motorista inválido/inativo: % / %', p_cnpj, p_motorista;
        RETURN json_build_object('error', 'Motorista inválido ou inativo');
    END IF;
    
    -- ==============================================
    -- 5. VERIFICAR SESSÃO EXISTENTE
    -- ==============================================
    SELECT * INTO v_sessao_existente
    FROM sessoes
    WHERE cnpj = p_cnpj 
      AND motorista = p_motorista 
      AND ativo = true;
    
    -- ==============================================
    -- 6. TRATAR SESSÃO EXISTENTE
    -- ==============================================
    IF FOUND THEN
        -- CASO A: Mesmo dispositivo → reaproveita sessão
        IF v_sessao_existente.device_id = p_device_id THEN
            UPDATE sessoes 
            SET ultimo_acesso = NOW()
            WHERE id = v_sessao_existente.id;
            
            RAISE LOG 'Sessão REAPROVEITADA - Motorista: %, Device: %, Token: %', 
              p_motorista, p_device_id, v_sessao_existente.token;
            
            SELECT COUNT(*) INTO v_sessoes_ativas
            FROM sessoes
            WHERE cnpj = p_cnpj AND ativo = true;
            
            RETURN json_build_object(
                'user', json_build_object(
                    'id', v_motorista.id,
                    'nome', v_motorista.nome,
                    'nome_empresa', v_empresa.nome_fantasia,
                    'motorista', p_motorista,
                    'cnpj', p_cnpj
                ),
                'token', v_sessao_existente.token,
                'license', json_build_object(
                    'limite', v_empresa.limite_usuarios_comprova,
                    'emUso', v_sessoes_ativas,
                    'expiraEm', to_char(v_empresa.comprova_expira_em, 'YYYY-MM-DD"T"HH24:MI:SS"Z"'),
                    'valida', true
                ),
                'reused', true
            );
        END IF;
        
        -- CASO B: Outro dispositivo → encerra sessão anterior
        UPDATE sessoes 
        SET ativo = false, 
            data_encerramento = NOW(),
            motivo_encerramento = 'substituída por outro dispositivo'
        WHERE id = v_sessao_existente.id;
        
        RAISE LOG 'Sessão ANTERIOR ENCERRADA - Motorista: %, Device Antigo: %, Device Novo: %', 
          p_motorista, v_sessao_existente.device_id, p_device_id;
    END IF;
    
    -- ==============================================
    -- 7. CONTAR SESSÕES ATIVAS
    -- ==============================================
    SELECT COUNT(*) INTO v_sessoes_ativas
    FROM sessoes
    WHERE cnpj = p_cnpj AND ativo = true;
    
    -- ==============================================
    -- 8. VALIDAR LIMITE DA LICENÇA
    -- ==============================================
    IF v_sessoes_ativas >= v_empresa.limite_usuarios_comprova THEN
        RAISE LOG 'Login falhou - Limite excedido: % / % ativas, limite %', 
          p_cnpj, v_sessoes_ativas, v_empresa.limite_usuarios_comprova;
        RETURN json_build_object(
            'error', 'Limite de usuários simultâneos atingido. Limite: ' || v_empresa.limite_usuarios_comprova
        );
    END IF;
    
    -- ==============================================
    -- 9. GERAR TOKEN E CRIAR NOVA SESSÃO
    -- ==============================================
    v_token := md5(p_device_id || p_cnpj || p_motorista || random()::text || clock_timestamp()::text);
    
    INSERT INTO sessoes (
        cnpj,
        motorista,
        device_id,
        token,
        ativo,
        data_login,
        ultimo_acesso,
        codfilial,
        ip,
        motivo_encerramento
    ) VALUES (
        p_cnpj,
        p_motorista,
        p_device_id,
        v_token,
        true,
        NOW(),
        NOW(),
        v_empresa.codfilial,
        'app_device',
        NULL
    );
    
    RAISE LOG 'Nova sessão CRIADA - Motorista: %, Device: %, Token: %', 
      p_motorista, p_device_id, v_token;
    
    -- ==============================================
    -- 10. CONTAGEM FINAL
    -- ==============================================
    SELECT COUNT(*) INTO v_sessoes_ativas
    FROM sessoes
    WHERE cnpj = p_cnpj AND ativo = true;
    
    -- ==============================================
    -- 11. RETORNAR RESULTADO
    -- ==============================================
    RETURN json_build_object(
        'user', json_build_object(
            'id', v_motorista.id,
            'nome', v_motorista.nome,
            'nome_empresa', v_empresa.nome_fantasia,
            'motorista', p_motorista,
            'cnpj', p_cnpj
        ),
        'token', v_token,
        'license', json_build_object(
            'limite', v_empresa.limite_usuarios_comprova,
            'emUso', v_sessoes_ativas,
            'expiraEm', to_char(v_empresa.comprova_expira_em, 'YYYY-MM-DD"T"HH24:MI:SS"Z"'),
            'valida', true
        ),
        'reused', false
    );
END;
$$;

3.2 Função validar_sessao()
sql
CREATE OR REPLACE FUNCTION validar_sessao(p_token TEXT)
RETURNS JSONB
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_sessao RECORD;
    v_timeout_minutes INTEGER := 30;
    v_motivo TEXT;
BEGIN
    SELECT * INTO v_sessao
    FROM sessoes
    WHERE token = p_token;
    
    -- Caso 1: Sessão não existe
    IF NOT FOUND THEN
        RETURN json_build_object('valida', false, 'motivo', 'sessao_nao_encontrada');
    END IF;
    
    -- Caso 2: Sessão está inativa
    IF NOT v_sessao.ativo THEN
        IF v_sessao.motivo_encerramento = 'substituída por outro dispositivo' THEN
            v_motivo := 'substituida_outro_dispositivo';
        ELSIF v_sessao.motivo_encerramento = 'Timeout de inatividade' THEN
            v_motivo := 'timeout';
        ELSIF v_sessao.motivo_encerramento = 'Logout manual' THEN
            v_motivo := 'logout_manual';
        ELSE
            v_motivo := 'sessao_encerrada';
        END IF;
        
        RETURN json_build_object(
            'valida', false, 
            'motivo', v_motivo,
            'detalhes', v_sessao.motivo_encerramento
        );
    END IF;
    
    -- Caso 3: Timeout por inatividade
    IF NOW() - v_sessao.ultimo_acesso > (v_timeout_minutes || ' minutes')::INTERVAL THEN
        UPDATE sessoes 
        SET ativo = false,
            data_encerramento = NOW(),
            motivo_encerramento = 'Timeout de inatividade'
        WHERE token = p_token;
        
        RETURN json_build_object(
            'valida', false, 
            'motivo', 'timeout',
            'detalhes', 'Sessão expirada por inatividade de ' || v_timeout_minutes || ' minutos'
        );
    END IF;
    
    -- Caso 4: Sessão válida
    UPDATE sessoes SET ultimo_acesso = NOW() WHERE token = p_token;
    
    RETURN json_build_object(
        'valida', true,
        'cnpj', v_sessao.cnpj,
        'motorista', v_sessao.motorista,
        'device_id', v_sessao.device_id
    );
END;
$$;
3.3 Função logout()

CREATE OR REPLACE FUNCTION logout(p_cnpj TEXT, p_motorista TEXT)
RETURNS VOID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
    UPDATE sessoes
    SET ativo = false,
        data_encerramento = NOW(),
        motivo_encerramento = 'Logout manual'
    WHERE cnpj = p_cnpj 
      AND motorista = p_motorista 
      AND ativo = true;
END;
$$;

3.6 Função get_current_cnpj() (para RLS)
sql
CREATE OR REPLACE FUNCTION get_current_cnpj()
RETURNS VARCHAR
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_cnpj VARCHAR;
BEGIN
    SELECT cnpj INTO v_cnpj
    FROM sessoes
    WHERE token = current_setting('app.current_token', true)
      AND ativo = true;
    
    RETURN v_cnpj;
END;
$$;
3.7 Função limpar_sessoes_inativas() (Manutenção)
sql
CREATE OR REPLACE FUNCTION limpar_sessoes_inativas()
RETURNS INTEGER
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_quantidade INTEGER;
BEGIN
    WITH updated AS (
        UPDATE sessoes 
        SET ativo = false,
            data_encerramento = NOW(),
            motivo_encerramento = 'Limpeza automática'
        WHERE ativo = true 
          AND ultimo_acesso < NOW() - INTERVAL '1 day'
        RETURNING id
    )
    SELECT COUNT(*) INTO v_quantidade FROM updated;
    
    RETURN v_quantidade;
END;
$$;

4.2 Políticas
sql
-- Empresas: usuário vê apenas sua empresa
CREATE POLICY "Usuarios veem apenas sua empresa" ON empresas
    FOR SELECT
    USING (cnpj = get_current_cnpj());

-- Motoristas: usuário vê motoristas da sua empresa
CREATE POLICY "Usuarios veem motoristas da sua empresa" ON motoristas
    FOR SELECT
    USING (cnpj = get_current_cnpj());

-- Sessões: usuário vê apenas sua própria sessão
CREATE POLICY "Usuarios veem apenas sua sessao" ON sessoes
    FOR SELECT
    USING (token = current_setting('app.current_token', true));

-- Sessões: usuário pode atualizar sua própria sessão
CREATE POLICY "Usuarios podem atualizar sua sessao" ON sessoes
    FOR UPDATE
    USING (token = current_setting('app.current_token', true))
    WITH CHECK (token = current_setting('app.current_token', true));

📋 8. REGRAS DE NEGÓCIO

### Regra C - Multi-device

Comportamento:
- Um mesmo motorista só pode ter 1 dispositivo ativo por vez
- A empresa só pode ter até o limite de sessões simultâneas definido pela licença
- Trocar de dispositivo encerra a sessão anterior do mesmo motorista
- Usar o mesmo dispositivo reaproveita a sessão existente
- Sessão inválida é detectada pelo app e encerra automaticamente

### Ordem de validação no login
1. validar empresa
2. validar status
3. validar licença (ativa e não expirada)
4. validar motorista
5. procurar sessão ativa do mesmo motorista
   - mesmo deviceId → reaproveita (atualiza ultimo_acesso)
   - outro deviceId → encerra a antiga (motivo: 'substituída por outro dispositivo')
6. contar sessões ativas da empresa
7. validar limite da licença
8. criar nova sessão

### Motivos de invalidação de sessão
- `sessao_nao_encontrada` - Token inválido
- `substituida_outro_dispositivo` - Motorista logou em outro aparelho
- `timeout` - Inatividade por 30 minutos
- `logout_manual` - Logout explícito
- `sessao_encerrada` - Outros motivos
